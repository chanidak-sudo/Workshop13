import 'package:flutter/material.dart';
import '../models/product_model.dart';

class CartProvider extends ChangeNotifier {
  final Map<String, int> _items = {};

  Map<String, int> get items => {..._items};

  int get totalItems =>
      _items.values.fold(0, (sum, qty) => sum + qty);

  double totalPrice(List<Product> catalog) {
    double total = 0;
    _items.forEach((id, qty) {
      final product = catalog.firstWhere((p) => p.id == id);
      total += product.price * qty;
    });
    return total;
  }

  double getVipDiscount(List<Product> catalog) {
    final raw = totalPrice(catalog);
    return raw >= 50000 ? raw * 0.1 : 0;
  }

  double getFinalTotal(List<Product> catalog) =>
      totalPrice(catalog) - getVipDiscount(catalog);

  void addToCart(String id) {
    _items[id] = (_items[id] ?? 0) + 1;
    notifyListeners();
  }

  void removeFromCart(String id) {
    if (!_items.containsKey(id)) return;
    if (_items[id] == 1) {
      _items.remove(id);
    } else {
      _items[id] = _items[id]! - 1;
    }
    notifyListeners();
  }

  void clearCart() {
    _items.clear();
    notifyListeners();
  }
}

📦 order_provider.dart

import 'package:flutter/material.dart';
import '../models/order_model.dart';

class OrderProvider extends ChangeNotifier {
  final List<OrderModel> _orders = [];

  List<OrderModel> get orders => [..._orders];

  void placeOrder(OrderModel order) {
    _orders.insert(0, order);
    notifyListeners();
    _simulateTracking(order.orderId);
  }

  void _simulateTracking(String id) {
    Future.delayed(const Duration(seconds: 10), () {
      final index = _orders.indexWhere((o) => o.orderId == id);
      if (index != -1) {
        _orders[index].status = OrderStatus.shipping;
        notifyListeners();
      }
    });

    Future.delayed(const Duration(seconds: 20), () {
      final index = _orders.indexWhere((o) => o.orderId == id);
      if (index != -1) {
        _orders[index].status = OrderStatus.delivered;
        notifyListeners();
      }
    });
  }
}

import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import '../models/product_model.dart';
import '../models/order_model.dart';
import '../providers/cart_provider.dart';
import '../providers/order_provider.dart';
import 'order_history_page.dart';

class CheckoutPage extends StatefulWidget {
  final List<Product> catalog;
  const CheckoutPage({super.key, required this.catalog});

  @override
  State<CheckoutPage> createState() => _CheckoutPageState();
}

class _CheckoutPageState extends State<CheckoutPage> {
  final _formKey = GlobalKey<FormState>();
  final _name = TextEditingController();
  final _address = TextEditingController();

  @override
  void dispose() {
    _name.dispose();
    _address.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final cart = Provider.of<CartProvider>(context, listen: false);
    final orderProvider = Provider.of<OrderProvider>(context, listen: false);

    return Scaffold(
      appBar: AppBar(
        title: const Text('ข้อมูลจัดส่งสินค้า'),
        backgroundColor: Colors.blueAccent,
        foregroundColor: Colors.white,
      ),
      body: Padding(
        padding: const EdgeInsets.all(20),
        child: Form(
          key: _formKey,
          child: ListView(
            children: [
              const Text('ชื่อผู้รับสินค้า'),
              TextFormField(
                controller: _name,
                validator: (v) => v == null || v.isEmpty ? 'กรุณากรอกชื่อ' : null,
              ),
              const SizedBox(height: 16),

              const Text('ที่อยู่จัดส่ง'),
              TextFormField(
                controller: _address,
                maxLines: 3,
                validator: (v) => v == null || v.isEmpty ? 'กรุณากรอกที่อยู่' : null,
              ),
              const SizedBox(height: 24),

              ElevatedButton(
                style: ElevatedButton.styleFrom(
                  minimumSize: const Size.fromHeight(50),
                  backgroundColor: Colors.blueAccent,
                  foregroundColor: Colors.white,
                ),
                onPressed: () {
                  if (!_formKey.currentState!.validate()) return;

                  final items = cart.items.entries.map((e) {
                    final p = widget.catalog.firstWhere((x) => x.id == e.key);
                    return OrderItem(product: p, quantity: e.value);
                  }).toList();

                  final order = OrderModel(
                    orderId: DateTime.now().millisecondsSinceEpoch.toString(),
                    items: items,
                    totalAmount: cart.getFinalTotal(widget.catalog),
                    customerName: _name.text,
                    customerAddress: _address.text,
                    orderDate: DateTime.now(),
                  );

                  orderProvider.placeOrder(order);
                  cart.clearCart();

                  Navigator.pushAndRemoveUntil(
                    context,
                    MaterialPageRoute(builder: (_) => const OrderHistoryPage()),
                    (route) => false,
                  );
                },
                child: const Text('ยืนยันการสั่งซื้อ'),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

class OrderHistoryPage extends StatelessWidget {
  const OrderHistoryPage({super.key});

  String statusText(OrderStatus s) {
    switch (s) {
      case OrderStatus.processing:
        return 'กำลังเตรียมสินค้า';
      case OrderStatus.shipping:
        return 'อยู่ระหว่างจัดส่ง';
      case OrderStatus.delivered:
        return 'จัดส่งสำเร็จ';

    switch (s) {
      case OrderStatus.processing:
        return Colors.orange;
      case OrderStatus.shipping:
        return Colors.blueAccent;
      case OrderStatus.delivered:
        return Colors.green;

        title: const Text('ประวัติคำสั่งซื้อ'),

import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import 'providers/cart_provider.dart';
import 'providers/order_provider.dart';
import 'pages/product_list_page.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MultiProvider(
      providers: [
        ChangeNotifierProvider(create: (_) => CartProvider()),
        ChangeNotifierProvider(create: (_) => OrderProvider()),
      ],
      child: MaterialApp(
        debugShowCheckedModeBanner: false,
        title: 'Shopping Cart Provider',
        theme: ThemeData(
          useMaterial3: true,
          colorSchemeSeed: Colors.blueAccent,
        ),
        home: ProductListPage(),
      ),
    );
  }
}
